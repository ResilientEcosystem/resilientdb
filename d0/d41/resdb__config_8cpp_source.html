42ResDBConfig::ResDBConfig(const ResConfigData& config_data,
43 const ReplicaInfo& self_info,
44 const KeyInfo& private_key,
45 const CertificateInfo& public_key_cert_info)
46 : config_data_(config_data),
47 self_info_(self_info),
48 private_key_(private_key),
49 public_key_cert_info_(public_key_cert_info) {
50 for (const auto& region : config_data.region()) {
51 if (region.region_id() == config_data.self_region_id()) {
52 LOG(INFO) << "get region info:" << region.DebugString();
53 for (const auto& replica : region.replica_info()) {
54 replicas_.push_back(replica);
55 }
56 LOG(INFO) << "get region config server size:"
57 << region.replica_info_size();
58 break;
59 }
60 }
61 if (config_data_.view_change_timeout_ms() == 0) {
62 config_data_.set_view_change_timeout_ms(viewchange_commit_timeout_ms_);
63 }
64 if (config_data_.client_batch_num() == 0) {
65 config_data_.set_client_batch_num(client_batch_num_);
66 }
67 if (config_data_.worker_num() == 0) {
68 config_data_.set_worker_num(worker_num_);
69 }
70 if (config_data_.input_worker_num() == 0) {
71 config_data_.set_input_worker_num(input_worker_num_);
72 }
73 if (config_data_.output_worker_num() == 0) {
74 config_data_.set_output_worker_num(output_worker_num_);
75 }
76 if (config_data_.tcp_batch_num() == 0) {
77 config_data_.set_tcp_batch_num(100);
78 }
79}

81void ResDBConfig::SetConfigData(const ResConfigData& config_data) {
82 config_data_ = config_data;
83 replicas_.clear();
84 for (const auto& region : config_data.region()) {
85 if (region.region_id() == config_data.self_region_id()) {
86 LOG(INFO) << "get region info:" << region.DebugString();
87 for (const auto& replica : region.replica_info()) {
88 replicas_.push_back(replica);
89 }
90 LOG(INFO) << "get region config server size:"
91 << region.replica_info_size();
92 break;
93 }
94 }
95 if (config_data_.view_change_timeout_ms() == 0) {
96 config_data_.set_view_change_timeout_ms(viewchange_commit_timeout_ms_);
97 }
98}
