65std::unique_ptr<std::string> KVServiceTransactionManager::ExecuteData(
66 const std::string& request) {
67 KVRequest kv_request;
68 KVResponse kv_response;
69
70 if (!kv_request.ParseFromString(request)) {
71 LOG(ERROR) << "parse data fail";
72 return nullptr;
73 }
74
75 if (kv_request.cmd() == KVRequest::SET) {
76 Set(kv_request.key(), kv_request.value());
77 } else if (kv_request.cmd() == KVRequest::GET) {
78 kv_response.set_value(Get(kv_request.key()));
79 } else if (kv_request.cmd() == KVRequest::GETVALUES) {
80 kv_response.set_value(GetValues());
81 } else if (kv_request.cmd() == KVRequest::GETRANGE) {
82 kv_response.set_value(GetRange(kv_request.key(), kv_request.value()));
83 }
84
85 std::unique_ptr<std::string> resp_str = std::make_unique<std::string>();
86 if (!kv_response.SerializeToString(resp_str.get())) {
87 return nullptr;
88 }
89
90 return resp_str;
91}
