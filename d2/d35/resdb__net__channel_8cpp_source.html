86// Connect to the server if not connected and send data.
87int ResDBNetChannel::SendFromKeepAlive(const std::string& data) {
88 for (int i = 0; i < max_retry_time_; ++i) {
89 if (!long_connecting_) {
90 if (Connect() == 0) {
91 long_connecting_ = true;
92 }
93 if (!long_connecting_) {
94 LOG(ERROR) << "connect fail:" << ip_ << " port:" << port_;
95 continue;
96 }
97 }
98 int ret = SendDataInternal(data);
99 if (ret >= 0) {
100 return ret;
101 }
102 long_connecting_ = false;
103 }
104 return -1;
105}

107int ResDBNetChannel::Send(const std::string& data) {
108 if (long_connect_tion_) {
109 return SendFromKeepAlive(data);
110 }
111 for (int i = 0; i < max_retry_time_; ++i) {
112 if (!connected_) {
113 if (Connect()) {
114 LOG(ERROR) << "connect fail:" << ip_ << " port:" << port_;
115 continue;
116 }
117 }
118 int ret = SendDataInternal(data);
119 if (ret >= 0) {
120 return ret;
121 } else if (ret == -1) {
122 LOG(ERROR) << "send data fail:" << ip_ << " port:" << port_;
123 }
124 }
125 return -1;
126}

138// Sign the message if verifier has been provied and send the message to the
139// server.
140std::string ResDBNetChannel::GetRawMessageString(
141 const google::protobuf::Message& message, SignatureVerifierInterface* verifier) {
142 ResDBMessage sig_message;
143 if (!message.SerializeToString(sig_message.mutable_data())) {
144 return "";
145 }
146
147 if (verifier != nullptr) {
148 auto signature_or = verifier->SignMessage(sig_message.data());
149 if (!signature_or.ok()) {
150 LOG(ERROR) << "Sign message fail";
151 return "";
152 }
153 sig_message.mutable_signature()->Swap(&(*signature_or));
154 } else {
155 // LOG(ERROR) << " no verifier";
156 }
157
158 std::string message_str;
159 if (!sig_message.SerializeToString(&message_str)) {
160 return "";
161 }
162 return message_str;
163}

173int ResDBNetChannel::SendRawMessage(const google::protobuf::Message& message) {
174 ResDBMessage sig_message;
175 if (!message.SerializeToString(sig_message.mutable_data())) {
176 return -1;
177 }
178
179 if (verifier_ != nullptr) {
180 auto signature_or = verifier_->SignMessage(sig_message.data());
181 if (!signature_or.ok()) {
182 LOG(ERROR) << "Sign message fail";
183 return -1;
184 }
185 sig_message.mutable_signature()->Swap(&(*signature_or));
186 }
187 std::string message_str;
188 if (!sig_message.SerializeToString(&message_str)) {
189 return -1;
190 }
191 return Send(message_str);
192}

194int ResDBNetChannel::RecvRawMessageStr(std::string* message) {
195 std::string recv_str;
196 int ret = Recv(&recv_str);
197 if (ret <= 0) {
198 return ret;
199 }
200 ResDBMessage sig_message;
201 if (!sig_message.ParseFromString(recv_str)) {
202 LOG(ERROR) << "parse to sig fail";
203 return -1;
204 }
205 *message = sig_message.data();
206 return 0;
207}

209int ResDBNetChannel::RecvRawMessage(google::protobuf::Message* message) {
210 std::string resp_data;
211 int ret = Recv(&resp_data);
212 if (ret < 0) {
213 LOG(ERROR) << "recv fail:" << ret;
214 return -1;
215 }
216 ResDBMessage sig_message;
217 if (!sig_message.ParseFromString(resp_data)) {
218 LOG(ERROR) << "parse sig msg fail";
219 return -1;
220 }
221 if (!message->ParseFromString(sig_message.data())) {
222 LOG(ERROR) << "parse response msg fail";
223 return -1;
224 }
225 return 0;
226}
