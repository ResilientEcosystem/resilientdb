47int ResDBTransactionProposer::SendRequest(
48 const google::protobuf::Message& message,
49 google::protobuf::Message* response, Request::Type type) {
50 ResDBNetChannel::SetDestReplicaInfo(config_.GetReplicaInfos()[0]);
51 int ret = ResDBNetChannel::SendRequest(message, type, true);
52 if (ret == 0) {
53 std::string resp_str;
54 int ret = ResDBNetChannel::RecvRawMessageData(&resp_str);
55 if (ret >= 0) {
56 if (!response->ParseFromString(resp_str)) {
57 LOG(ERROR) << "parse response fail:" << resp_str.size();
58 return -2;
59 }
60 return 0;
61 }
62 }
63 return -1;
64}
