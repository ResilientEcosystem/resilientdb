35std::unique_ptr<std::string> KVExecutor::ExecuteData(
36 const std::string& request) {
37 KVRequest kv_request;
38 KVResponse kv_response;
39
40 if (!kv_request.ParseFromString(request)) {
41 LOG(ERROR) << "parse data fail";
42 return nullptr;
43 }
44
45 if (kv_request.cmd() == KVRequest::SET) {
46 Set(kv_request.key(), kv_request.value());
47 } else if (kv_request.cmd() == KVRequest::GET) {
48 kv_response.set_value(Get(kv_request.key()));
49 } else if (kv_request.cmd() == KVRequest::GETALLVALUES) {
50 kv_response.set_value(GetAllValues());
51 } else if (kv_request.cmd() == KVRequest::GETRANGE) {
52 kv_response.set_value(GetRange(kv_request.key(), kv_request.value()));
53 } else if (kv_request.cmd() == KVRequest::SET_WITH_VERSION) {
54 SetWithVersion(kv_request.key(), kv_request.value(), kv_request.version());
55 } else if (kv_request.cmd() == KVRequest::GET_WITH_VERSION) {
56 GetWithVersion(kv_request.key(), kv_request.version(),
57 kv_response.mutable_value_info());
58 } else if (kv_request.cmd() == KVRequest::GET_ALL_ITEMS) {
59 GetAllItems(kv_response.mutable_items());
60 } else if (kv_request.cmd() == KVRequest::GET_KEY_RANGE) {
61 GetKeyRange(kv_request.min_key(), kv_request.max_key(),
62 kv_response.mutable_items());
63 } else if (kv_request.cmd() == KVRequest::GET_HISTORY) {
64 GetHistory(kv_request.key(), kv_request.min_version(),
65 kv_request.max_version(), kv_response.mutable_items());
66 } else if (kv_request.cmd() == KVRequest::GET_TOP) {
67 GetTopHistory(kv_request.key(), kv_request.top_number(),
68 kv_response.mutable_items());
69 }
70
71 std::unique_ptr<std::string> resp_str = std::make_unique<std::string>();
72 if (!kv_response.SerializeToString(resp_str.get())) {
73 return nullptr;
74 }
75 return resp_str;
76}
